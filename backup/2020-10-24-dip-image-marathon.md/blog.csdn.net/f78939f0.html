OpenCV图像修复

在OpenCV的“photo.hpp”中定义了一个inpaint函数，可以用来实现图像的修复和复原功能，inpaint函数的原型如下：

  1. void inpaint( InputArray src, InputArray inpaintMask,
  2.                            OutputArray dst, double inpaintRadius, int flags );

第一个参数src，输入的单通道或三通道图像；
第二个参数inpaintMask，图像的掩码，单通道图像，大小跟原图像一致，inpaintMask图像上除了需要修复的部分之外其他部分的像素值全部为0；
第三个参数dst，输出的经过修复的图像；
第四个参数inpaintRadius，修复算法取的邻域半径，用于计算当前像素点的差值；
第五个参数flags，修复算法，有两种：INPAINT_NS 和I NPAINT_TELEA；

函数实现关键是图像掩码的确定，可以通过阈值筛选或者手工选定，按照这个思路，用三种方法生成掩码，对比图像修复的效果。


方法一、全区域阈值处理+Mask膨胀处理
 

  1. #include <imgproc\imgproc.hpp>
  2. #include <highgui\highgui.hpp>
  3. #include <photo\photo.hpp>
  4.  
  5. using namespace cv;
  6.  
  7. //全区域阈值处理+Mask膨胀处理
  8. int main()
  9. {
  10. 	Mat imageSource = imread("Test.jpg");
  11. 	if (!imageSource.data)
  12. 	{
  13. 		return -1;
  14. 	}
  15. 	imshow("原图", imageSource);
  16. 	Mat imageGray;
  17. 	//转换为灰度图
  18. 	cvtColor(imageSource, imageGray, CV_RGB2GRAY, 0);
  19. 	Mat imageMask = Mat(imageSource.size(), CV_8UC1, Scalar::all(0));
  20.  
  21. 	//通过阈值处理生成Mask
  22. 	threshold(imageGray, imageMask, 240, 255, CV_THRESH_BINARY);
  23. 	Mat Kernel = getStructuringElement(MORPH_RECT, Size(3, 3));
  24. 	//对Mask膨胀处理，增加Mask面积
  25. 	dilate(imageMask, imageMask, Kernel);
  26.  
  27. 	//图像修复
  28. 	inpaint(imageSource, imageMask, imageSource, 5, INPAINT_TELEA);
  29. 	imshow("Mask", imageMask);
  30. 	imshow("修复后", imageSource);
  31. 	waitKey();
  32. }

方法二、鼠标框选区域+阈值处理+Mask膨胀处理


  1. #include <imgproc/imgproc.hpp>
  2. #include <highgui/highgui.hpp>
  3. #include <core/core.hpp>
  4. #include <photo/photo.hpp>
  5.  
  6. using namespace cv;
  7.  
  8. Point ptL, ptR; //鼠标画出矩形框的起点和终点
  9. Mat imageSource, imageSourceCopy;
  10. Mat ROI; //原图需要修复区域的ROI
  11.  
  12. //鼠标回调函数
  13. void OnMouse(int event, int x, int y, int flag, void *ustg);
  14.  
  15. //鼠标圈定区域阈值处理+Mask膨胀处理
  16. int main()
  17. {
  18. 	imageSource = imread("Test.jpg");
  19. 	if (!imageSource.data)
  20. 	{
  21. 		return -1;
  22. 	}
  23. 	imshow("原图", imageSource);
  24. 	setMouseCallback("原图", OnMouse);
  25. 	waitKey();
  26. }
  27. void OnMouse(int event, int x, int y, int flag, void *ustg)
  28. {
  29. 	if (event == CV_EVENT_LBUTTONDOWN)
  30. 	{
  31. 		ptL = Point(x, y);
  32. 		ptR = Point(x, y);
  33. 	}
  34. 	if (flag == CV_EVENT_FLAG_LBUTTON)
  35. 	{
  36. 		ptR = Point(x, y);
  37. 		imageSourceCopy = imageSource.clone();
  38. 		rectangle(imageSourceCopy, ptL, ptR, Scalar(255, 0, 0));
  39. 		imshow("原图", imageSourceCopy);
  40. 	}
  41. 	if (event == CV_EVENT_LBUTTONUP)
  42. 	{
  43. 		if (ptL != ptR)
  44. 		{
  45. 			ROI = imageSource(Rect(ptL, ptR));
  46. 			imshow("ROI", ROI);
  47. 			waitKey();
  48. 		}
  49. 	}
  50. 	//单击鼠标右键开始图像修复
  51. 	if (event == CV_EVENT_RBUTTONDOWN)
  52. 	{
  53. 		imageSourceCopy = ROI.clone();
  54. 		Mat imageGray;
  55. 		cvtColor(ROI, imageGray, CV_RGB2GRAY); //转换为灰度图
  56. 		Mat imageMask = Mat(ROI.size(), CV_8UC1, Scalar::all(0));
  57.  
  58. 		//通过阈值处理生成Mask
  59. 		threshold(imageGray, imageMask, 235, 255, CV_THRESH_BINARY);
  60. 		Mat Kernel = getStructuringElement(MORPH_RECT, Size(3, 3));
  61. 		dilate(imageMask, imageMask, Kernel);  //对Mask膨胀处理
  62. 		inpaint(ROI, imageMask, ROI, 9, INPAINT_TELEA);  //图像修复
  63. 		imshow("Mask", imageMask);
  64. 		imshow("修复后", imageSource);
  65. 	}
  66. }

方法三、鼠标划定整个区域作为修复对象


这个方法选定一个矩形区域，把整个矩形区域作为要修复的对象，该方法适用于图像结构比较简单，特别是纯色图像，并且选定区域面积占比不大的情况，效果较好。

  1. #include <imgproc/imgproc.hpp>
  2. #include <highgui/highgui.hpp>
  3. #include <core/core.hpp>
  4. #include <photo/photo.hpp>
  5.  
  6. using namespace cv;
  7.  
  8. Point ptL, ptR; //鼠标画出矩形框的起点和终点
  9. Mat imageSource, imageSourceCopy;
  10. Mat ROI; //原图需要修复区域的ROI
  11.  
  12. //鼠标回调函数
  13. void OnMouse(int event, int x, int y, int flag, void *ustg);
  14.  
  15. //鼠标圈定区域
  16. int main()
  17. {
  18. 	imageSource = imread("Test.jpg");
  19. 	if (!imageSource.data)
  20. 	{
  21. 		return -1;
  22. 	}
  23. 	imshow("原图", imageSource);
  24. 	setMouseCallback("原图", OnMouse);
  25. 	waitKey();
  26. }
  27. void OnMouse(int event, int x, int y, int flag, void *ustg)
  28. {
  29. 	if (event == CV_EVENT_LBUTTONDOWN)
  30. 	{
  31. 		ptL = Point(x, y);
  32. 		ptR = Point(x, y);
  33. 	}
  34. 	if (flag == CV_EVENT_FLAG_LBUTTON)
  35. 	{
  36. 		ptR = Point(x, y);
  37. 		imageSourceCopy = imageSource.clone();
  38. 		rectangle(imageSourceCopy, ptL, ptR, Scalar(255, 0, 0));
  39. 		imshow("原图", imageSourceCopy);
  40. 	}
  41. 	if (event == CV_EVENT_LBUTTONUP)
  42. 	{
  43. 		if (ptL != ptR)
  44. 		{
  45. 			ROI = imageSource(Rect(ptL, ptR));
  46. 			imshow("ROI", ROI);
  47. 			waitKey();
  48. 		}
  49. 	}
  50. 	//单击鼠标右键开始图像修复
  51. 	if (event == CV_EVENT_RBUTTONDOWN)
  52. 	{
  53. 		imageSourceCopy = Mat(imageSource.size(), CV_8UC1, Scalar::all(0));
  54. 		Mat imageMask = imageSourceCopy(Rect(ptL, ptR));
  55. 		//生成一个跟ROI大小一样的值全为1的区域
  56. 		Mat imageMaskCopy = Mat(imageMask.size(), CV_8UC1, Scalar::all(1));
  57. 		imageMaskCopy.copyTo(imageMask);
  58. 		inpaint(imageSource, imageSourceCopy, imageSource, 9, INPAINT_TELEA);  //图像修复
  59. 		imshow("Mask", imageSourceCopy);
  60. 		imshow("修复后", imageSource);
  61. 	}
  62. }
