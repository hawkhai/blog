CreateMapFileAndGetBaseAddr()的定义就看我下一篇文章。

  5. #define NB10_SIG	'01BN'
  6. #define RSDS_SIG	'SDSR'
  7. typedef struct  CV_HEADER
  8. {
  9. 	DWORD Signature;
  10. 	DWORD Offset;
  11. }CV_HEADER;
  12.  
  13. typedef struct CV_INFO_PDB20
  14. {
  15. 	CV_HEADER	CvHeader;
  16. 	DWORD		Signature;
  17. 	DWORD		Age;
  18. 	BYTE		PdbFileName[1];
  19. }CV_INFO_PDB20;
  20. typedef struct CV_INFO_PDB70
  21. {
  22. 	DWORD	CvSignature;
  23. 	GUID	Signature;
  24. 	DWORD	Age;
  25. 	BYTE	PdbFileName[1];
  26. }CV_INFO_PDB70;

  28. BOOL PeIsRegionValid(PVOID Base, DWORD Size, PVOID Addr, DWORD RegionSize)
  29. {
  30. 	return ((PBYTE)Addr >= (PBYTE)Base && ((PBYTE)Addr + RegionSize) <= ((PBYTE)Base + Size));
  31. }
  32.  
  33. //返回TRUE 但PdbStr为空意思是找过了确实没有PDB
  34. BOOLEAN PeGetPdb(PVOID ImageBase, DWORD ImageSize, PCHAR PdbStr)
  35. {
  36. 	PBYTE Base = (PBYTE)ImageBase;
  37. 	BOOLEAN Result = FALSE;
  38. 	CV_HEADER* CvInfo;
  39. 	PIMAGE_DEBUG_DIRECTORY DbgDir;
  40. 	ULONG PdbNameSize = 0;
  41.  
  42. 	__try
  43. 	{
  44. 		do
  45. 		{
  46. 			PIMAGE_DOS_HEADER DosHeader = (PIMAGE_DOS_HEADER)Base;
  47. 			PIMAGE_NT_HEADERS NtHeader = (PIMAGE_NT_HEADERS)(Base + DosHeader->e_lfanew);
  48. 			PIMAGE_OPTIONAL_HEADER		OptionalHeader;
  49. 			PIMAGE_OPTIONAL_HEADER64	OptionalHeader64;
  50. 			PIMAGE_OPTIONAL_HEADER32	OptionalHeader32;
  51. 			DWORD DbgDirRva = 0;
  52.  
  53. 			//PE解析代码
  54. 			OptionalHeader = (PIMAGE_OPTIONAL_HEADER)(Base + DosHeader->e_lfanew + FIELD_OFFSET(IMAGE_NT_HEADERS, OptionalHeader));
  55. 			if (OptionalHeader->Magic == IMAGE_NT_OPTIONAL_HDR64_MAGIC)	// PE32+  x64
  56. 			{
  57. 				OptionalHeader64 = (PIMAGE_OPTIONAL_HEADER64)OptionalHeader;
  58. 				DbgDirRva = OptionalHeader64->DataDirectory[IMAGE_DIRECTORY_ENTRY_DEBUG].VirtualAddress;
  59. 			}
  60. 			else	// PE32   x86
  61. 			{
  62. 				OptionalHeader32 = (PIMAGE_OPTIONAL_HEADER32)OptionalHeader;
  63. 				DbgDirRva = OptionalHeader32->DataDirectory[IMAGE_DIRECTORY_ENTRY_DEBUG].VirtualAddress;
  64. 			}
  65.  
  66. 			DbgDir = (PIMAGE_DEBUG_DIRECTORY)(Base + DbgDirRva);
  67. 			if (!DbgDir)
  68. 				break;
  69.  
  70. 			ProbeForRead(DbgDir, sizeof(IMAGE_DEBUG_DIRECTORY), 1);
  71.  
  72. 			Result = TRUE;//到此表示解析正常 能不能找到PDB就看有没有了
  73.  
  74. 			if (!DbgDir->AddressOfRawData || DbgDir->Type != IMAGE_DEBUG_TYPE_CODEVIEW)
  75. 				break;
  76.  
  77. 			CvInfo = (CV_HEADER*)(Base + DbgDir->AddressOfRawData);
  78. 			if (!PeIsRegionValid(Base, ImageSize, CvInfo, sizeof(CV_HEADER)))
  79. 				break;
  80.  
  81. 			if (CvInfo->Signature == NB10_SIG) //VC6.0 (GBK)
  82. 			{
  83. 				if (!PeIsRegionValid(Base, ImageSize, CvInfo, sizeof(CV_INFO_PDB20)+MAX_PATH))
  84. 					break;
  85.  
  86. 				PdbNameSize = strlen((CHAR*)((CV_INFO_PDB20*)CvInfo)->PdbFileName);
  87. 				if (!PdbNameSize || PdbNameSize >= MAX_PATH)
  88. 					break;
  89.  
  90. 				RtlCopyMemory(PdbStr, (CHAR*)((CV_INFO_PDB20*)CvInfo)->PdbFileName, PdbNameSize);
  91. 			}
  92. 			else if (CvInfo->Signature == RSDS_SIG) //VS2003+ (UTF-8)
  93. 			{
  94. 				if (!PeIsRegionValid(Base, ImageSize, CvInfo, sizeof(CV_INFO_PDB70)+MAX_PATH))
  95. 					break;
  96.  
  97. 				PdbNameSize = strlen((CHAR*)((CV_INFO_PDB70*)CvInfo)->PdbFileName);
  98. 				if (!PdbNameSize || PdbNameSize >= MAX_PATH)
  99. 					break;
  100.  
  101. 				RtlCopyMemory(PdbStr, (CHAR*)((CV_INFO_PDB70*)CvInfo)->PdbFileName, PdbNameSize);
  102. 			}
  103. 		} while (0);
  104.  
  105. 	}
  106. 	__except (EXCEPTION_EXECUTE_HANDLER)
  107. 	{
  108. 		KdPrint(("EXCEPTION_EXECUTE_HANDLER\n"));
  109. 		Result = FALSE;
  110. 	}
  111.  
  112. 	return Result;
  113. }
  114.  
  115.  
  116. BOOLEAN PeGetFilePdb(PUNICODE_STRING pFilePath, PCHAR PdbStr)
  117. {
  118. 	BOOLEAN Result = FALSE;
  119. 	SIZE_T Size = 0;
  120. 	PVOID BaseAddress = CreateMapFileAndGetBaseAddr(pFilePath, &Size);
  121. 	if (NULL == BaseAddress)
  122. 		return FALSE;
  123.  
  124. 	Result = PeGetPdb(BaseAddress, Size, PdbStr);
  125.  
  126. 	ZwUnmapViewOfSection(ZwCurrentProcess(), BaseAddress);
  127. 	return Result;
  128. }
