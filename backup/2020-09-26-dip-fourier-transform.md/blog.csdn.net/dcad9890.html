       实现的代码如下，图像还是那熟悉的动漫女头，处理后边缘显得很清晰很突出，而除了边缘外其他的区域的颜色都是浅色。当sigma的数值越大，边缘提取得越精确，但是有可能出现不完整的边缘信息，即边缘不连续，所以也不适合取值过大。

  1. import cv2
  2. import numpy as np
  3.  
  4. def GaussianHighFrequencyFilter(src, dst, sigma = 1):
  5.  
  6. 	im = cv2.imread(src, 0)
  7. 	imarr = np.array(im)
  8.  
  9. 	height, width = imarr.shape
  10.  
  11. 	fft = np.fft.fft2(imarr)
  12. 	fft = np.fft.fftshift(fft)
  13.  
  14. 	for i in range(height):
  15. 		for j in range(height):
  16. 			fft[i, j] *= (1 - np.exp(-((i - (height - 1)/2)**2 + (j - (width - 1)/2)**2)/2/sigma**2))
  17.  
  18. 	fft = np.fft.ifftshift(fft)
  19. 	ifft = np.fft.ifft2(fft)
  20.  
  21. 	ifft = np.real(ifft)
  22. 	max = np.max(ifft)
  23. 	min = np.min(ifft)
  24.  
  25. 	res = np.zeros((height, width), dtype = "uint8")
  26.  
  27. 	for i in range(height):
  28. 		for j in range(width):
  29. 			res[i, j] = 255 * (ifft[i, j] - min)/(max - min)
  30.  
  31. 	cv2.imwrite(dst, res)
