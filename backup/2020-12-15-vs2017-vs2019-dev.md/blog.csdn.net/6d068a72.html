这里该有的都有了，把这份软链到正确的地方，就好了。

cp /usr/local/lib64/libstdc++.so.6.0.21 /usr/lib64/
cd /usr/lib64/
rm -f libstdc++.so.6
ln -s libstdc++.so.6.0.21 libstdc++.so.6

./bootstrap
gmake
make install
