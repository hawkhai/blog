Irrlicht学习笔记(10)--PerPixelLighting

说明:

本例子介绍:
逐像素光照
雾
移动粒子.
使用引擎内置的一个更复杂的材质shader: 使用法线图和视差(parallax)图实现逐像素光照表面
详细步骤,请看源码注释

源码:

  1. /**
  2. *逐像素光照,雾,移动光,粒子
  3. */
  4. #include <iostream>
  5. #include <irrlicht.h>
  6. #include "driverChoice.h"
  7.  
  8. using namespace irr;
  9.  
  10. #ifdef _IRR_WINDOWS_
  11. #pragma comment(lib, "irrlicht.lib")
  12. //#pragma comment(linker, "/subsystem:windows /ENTRY:mainCRTStartup")
  13. #endif
  14.  
  15.  
  16. /**
  17. *1.写一个事件接收器
  18. *作用:用户选择当前材质种类,创建小的GUI窗口显示当前被选用的材质
  19. */
  20. class MyEventReceiver :public IEventReceiver
  21. {
  22. private:
  23. 	/**
  24. 	*1.1四个变量
  25. 	*/
  26. 	gui::IGUIStaticText* ProblemText;//显示错误内容
  27. 	gui::IGUIListBox* ListBox;//列表盒,用于展示所有可选内容
  28.  
  29. 	scene::ISceneNode* Room;//保存场景节点
  30. 	video::IVideoDriver* Driver;//保存driver
  31.  
  32. 	/**
  33. 	*1.2方法
  34. 	*二个共有方法:初始化,事件处理函数
  35. 	*一个私有方法:设置材质
  36. 	*/
  37. public:
  38. 	//1.2.1构造函数,初始化
  39. 	MyEventReceiver(scene::ISceneNode* room, 
  40. 		gui::IGUIEnvironment* env, video::IVideoDriver* driver)
  41. 	{
  42. 		//1.2.1.1这两个变量用于改变绘制模式
  43. 		Room = room;
  44. 		Driver = driver;
  45. 		//1.2.1.2添加listBox
  46. 		//调整字体
  47. 		gui::IGUISkin* skin = env->getSkin();
  48. 		gui::IGUIFont* font = env->getFont("../media/fonthaettenschweiler.bmp");
  49. 		if (font)
  50. 			skin->setFont(font);
  51.  
  52. 		//添加窗口,然后添加列表盒
  53. 		gui::IGUIWindow* window = env->addWindow(
  54. 			core::rect<s32>(0, 375, 160, 470), false, L"Use 'E'+'R' to change");
  55. 		ListBox = env->addListBox(
  56. 			core::rect<s32>(2, 32, 165, 88), window);
  57. 		//添加内容
  58. 		ListBox->addItem(L"Diffuse");//0
  59. 		ListBox->addItem(L"Dump mapping");//1
  60. 		ListBox->addItem(L"Parallax mapping");//2
  61.  
  62. 		//1.2.1.3创建错误文本
  63. 		ProblemText = env->addStaticText(
  64. 			L"Your hardware or this renderer is not able to use the "\
  65. 			L"needed shaders for this material. Using fall back materials.",
  66. 			core::rect<s32>(150, 20, 570, 80));
  67. 		ProblemText->setOverrideColor(video::SColor(100, 255, 255, 255));
  68.  
  69. 		//1.2.1.4设置初始材质
  70. 		//IMatterialRenderer是材质渲染接口,可以通过它进行扩展
  71. 		video::IMaterialRenderer* renderer = Driver->getMaterialRenderer(
  72. 			video::EMT_PARALLAX_MAP_SOLID);
  73. 		if (renderer&&renderer->getRenderCapability() == 0)
  74. 			ListBox->setSelected(2);//selected =2
  75. 	}
  76. 	//1.2.2事件处理函数
  77. 	bool OnEvent(const SEvent& event)
  78. 	{
  79. 		//检查用户是否按下E和R
  80. 		if (event.EventType == irr::EET_KEY_INPUT_EVENT&&
  81. 			!event.KeyInput.PressedDown&&Room&&ListBox)
  82. 		{
  83. 			//改变选择的对象
  84. 			int sel = ListBox->getSelected();
  85. 			if (event.KeyInput.Key == irr::KEY_KEY_R)
  86. 				++sel;
  87. 			else if (event.KeyInput.Key == irr::KEY_KEY_E)
  88. 				--sel;
  89. 			else
  90. 				return false;
  91.  
  92. 			if (sel > 2)sel = 0;
  93. 			if (sel < 0)sel = 2;
  94. 			ListBox->setSelected(sel);
  95.  
  96. 			setMaterial();
  97. 		}
  98. 		return false;//error!如果不加这行,将不能处理其他消息
  99. 	}
  100. private:
  101. 	//1.2.3设置材质
  102. 	void setMaterial()
  103. 	{
  104. 		//使用当前列表盒选用的材质设置房间
  105. 		video::E_MATERIAL_TYPE type = video::EMT_SOLID;
  106.  
  107. 		//改变材质设置
  108. 		switch (ListBox->getSelected())
  109. 		{
  110. 		case 0:
  111. 			type = video::EMT_SOLID;
  112. 			break;
  113. 		case 1:
  114. 			/**
  115. 			*实体法线图渲染器
  116. 			*第一个图为纹理图,第二个图为法线图
  117. 			*只有当顶点是S3DVertexTangents(EVT_TANGENTS)格式才可以使用这种材质
  118. 			*可以使用IMeshManipulator::createMeshWithTangent()将任何网格转换成所需格式
  119. 			*/
  120. 			type = video::EMT_NORMAL_MAP_SOLID;
  121. 			break;
  122. 		case 2:
  123. 			/**
  124. 			*和EMT_NORMAL_MAP_SOLID差不多,只是只有的是parallax图
  125. 			*更加真实
  126. 			*需要硬件支持:顶点着色器1.1,片元着色器1.4
  127. 			*第一个图为纹理图,第二个图为法线图(32bits)(高度图)
  128. 			*法线图纹理需要在其透明度通道中包含高度信息
  129. 			*高度信息可以通过函数IVideoDriver::makeNormalMapTexture自动设置,
  130. 			*函数参数:第一个:纹理图(其alpha通道可改变),第二个:amplitude,缩放比例
  131. 			*/
  132. 			type = video::EMT_PARALLAX_MAP_SOLID;
  133. 			break;
  134. 		}
  135. 		Room->setMaterialType(type);
  136. 		/*
  137. 		*如果材质不能正常显示,需要发出警告信息,
  138. 		*检查硬件是否支持:
  139. 		*IMaterialRenderer::getRenderCapability(),返回0就支持
  140. 		*/
  141. 		video::IMaterialRenderer* renderer = Driver->getMaterialRenderer(type);
  142. 		if (!renderer || renderer->getRenderCapability() != 0)
  143. 			ProblemText->setVisible(true);
  144. 		else
  145. 			ProblemText->setVisible(false);
  146. 	}
  147. };
  148. int main(int argc, char** argv)
  149. {
  150. 	video::E_DRIVER_TYPE driverType = driverChoiceConsole();
  151. 	IrrlichtDevice *device =
  152. 		createDevice(driverType, core::dimension2d<u32>(720, 455), 32,
  153. 		false, true, false, 0);
  154. 	if (!device)
  155. 		return 1;
  156. 	device->setWindowCaption(L"11PerPixelLighting");
  157.  
  158. 	video::IVideoDriver *driver = device->getVideoDriver();
  159. 	scene::ISceneManager *smgr = device->getSceneManager();
  160. 	gui::IGUIEnvironment *guiev = device->getGUIEnvironment();
  161. 	/*
  162. 	*2.准备工作:
  163. 	*加载logo,设置FPS摄像机,让引擎始终创建32位纹理,
  164. 	*最后一项是parallax图所必须的
  165. 	*/
  166. 	driver->setTextureCreationFlag(video::ETCF_ALWAYS_32_BIT, true);
  167. 	guiev->addImage(
  168. 		driver->getTexture("../media/irrlichtlogo2.png"),
  169. 		core::position2di(10, 10));
  170. 	scene::ICameraSceneNode* camera = smgr->addCameraSceneNodeFPS();
  171. 	camera->setPosition(core::vector3df(-200, 200, -200));
  172. 	device->getCursorControl()->setVisible(false);
  173.  
  174. 	/*
  175. 	*3.添加雾.
  176. 	*有多种雾,这里是使用pixel fog,配合将使用的材质
  177. 	*对每一个场景节点,只有将其EMF_FOG_ENABLE置true,才会被雾影响
  178. 	*/
  179. 	driver->setFog(
  180. 		video::SColor(0, 138, 125, 81),//color
  181. 		video::EFT_FOG_LINEAR,//fogType
  182. 		250,//start
  183. 		1000,//end
  184. 		0.003,//density
  185. 		true,//pixelFog
  186. 		false//rangeFog
  187. 		);
  188.  
  189. 	/*
  190. 	*4.加载房间
  191. 	*修复纹理:ImeshManipulator::makePlanarTextureMapping()
  192. 	*/
  193. 	scene::IAnimatedMesh* roomMesh = smgr->getMesh("../media/room.3ds");
  194. 	scene::ISceneNode* room = 0;
  195. 	if (roomMesh)
  196. 	{
  197. 		smgr->getMeshManipulator()->makePlanarTextureMapping(roomMesh->getMesh(0), 0.003f);
  198. 		/*
  199. 		*4.1
  200. 		*贴纹理图,
  201. 		*法线图作为第二层图,
  202. 		*法线图由加载进来的灰度纹理形式的高度图生成
  203. 		*通常法线图可以用自己创建的法线图,
  204. 		*这里用VideoDriver::makeNormalMapTexture产生
  205. 		*/
  206. 		video::ITexture* normalMap =
  207. 			driver->getTexture("../media/rockwall_height.bmp");
  208. 		if (normalMap)
  209. 			driver->makeNormalMapTexture(normalMap, 9.0f/*越大越凹凸*/);
  210. 		/*
  211. 		*4.2
  212. 		*所要使用的材质需要其他信息,比如:
  213. 		*tangents,binormals
  214. 		*所以需要计算这些信息.
  215. 		*使用:IMeshManipulator::createMeshWithTangents()创建
  216. 		*它从其他网格拷贝创建带有切线tangents和副法线binormals的网格
  217. 		*然后需要用这个网格创建一个标注的网格
  218. 		*然后就可以设置其他参宿了
  219. 		*/
  220. 		scene::IMesh* tangentMesh = smgr->getMeshManipulator()->
  221. 			createMeshWithTangents(roomMesh->getMesh(0));
  222. 		room = smgr->addMeshSceneNode(tangentMesh);
  223. 		room->setMaterialTexture(0, driver->getTexture("../media/rockwall.jpg"));
  224. 		room->setMaterialTexture(1, normalMap);
  225. 		room->getMaterial(0).SpecularColor.set(0, 0, 0, 0);
  226. 		room->setMaterialFlag(video::EMF_FOG_ENABLE, true);
  227. 		room->getMaterial(0).MaterialTypeParam = 0.035f;//自由参数
  228.  
  229. 		tangentMesh->drop();
  230. 	}
  231. 	/*
  232. 	*5.添加一个球
  233. 	*/
  234.  
  235.  
  236. 	/*
  237. 	*6.添加移动光源
  238. 	*逐像素光照在移动光照下才能凸显
  239. 	*这里添加两个移动光源,都绑定一个绕圈的动画和一个布告板
  240. 	*其中一个绑定一个粒子系统
  241. 	*/
  242.  
  243. 	/*
  244. 	*6.1.红色灯光
  245. 	*/
  246. 	scene::ILightSceneNode* light1 =
  247. 		smgr->addLightSceneNode(0, core::vector3df(0, 0, 0),
  248. 		video::SColorf(0.5f, 1.0f, 0.5f, 0.0f), 800.0f);
  249. 	light1->setDebugDataVisible(scene::EDS_BBOX);//显示包围盒
  250. 	//动画
  251. 	scene::ISceneNodeAnimator* anim = smgr->createFlyCircleAnimator(
  252. 		core::vector3df(50, 300, 0), 190, -0.003f);//中心点,半径,速度
  253. 	light1->addAnimator(anim);
  254. 	anim->drop();
  255. 	//布告板
  256. 	scene::ISceneNode* bill = smgr->addBillboardSceneNode(
  257. 		light1, core::dimension2d<f32>(60, 60));
  258. 	bill->setMaterialFlag(video::EMF_LIGHTING, false);
  259. 	bill->setMaterialFlag(video::EMF_ZWRITE_ENABLE, false);//不添加到z缓存
  260. 	bill->setMaterialType(video::EMT_TRANSPARENT_ADD_COLOR);
  261. 	bill->setMaterialTexture(0, driver->getTexture("../media/particlered.bmp"));
  262.  
  263.  
  264. 	/*
  265. 	*6.2.light2
  266. 	*额外:添加粒子系统,
  267. 	*由于这个版本(irr1.7.3)的这个材质shader使用的是ps1.1,vs1.1所以只能添加两个灯光
  268. 	*/
  269. 	scene::ILightSceneNode* light2 =
  270. 		smgr->addLightSceneNode(0, core::vector3df(0, 0, 0),
  271. 		video::SColorf(1.0f, 0.2f, 0.2f, 0.0f), 800.0f);
  272. 	//动画
  273. 	anim = smgr->createFlyCircleAnimator(
  274. 		core::vector3df(0, 150, 0), 200, 0.001f,core::vector3df(0.2f,0.9f,0.f));//中心点,半径,速,方向
  275. 	light2->addAnimator(anim);
  276. 	anim->drop();
  277. 	//布告板
  278. 	bill = smgr->addBillboardSceneNode(
  279. 		light2, core::dimension2d<f32>(120, 120));
  280. 	bill->setMaterialFlag(video::EMF_LIGHTING, false);
  281. 	bill->setMaterialFlag(video::EMF_ZWRITE_ENABLE, false);//不添加到z缓存
  282. 	bill->setMaterialType(video::EMT_TRANSPARENT_ADD_COLOR);
  283. 	bill->setMaterialTexture(0, driver->getTexture("../media/particlewhite.bmp"));
  284. 	//添加粒子系统
  285. 	scene::IParticleSystemSceneNode* ps =
  286. 		smgr->addParticleSystemSceneNode(false, light2);
  287. 	//发射器
  288. 	scene::IParticleEmitter* em = ps->createBoxEmitter(
  289. 		core::aabbox3d<f32>(-3, 0, -3, 3, 1, 3),
  290. 		core::vector3df(0.0f, 0.03f, 0.0f),
  291. 		80, 100,
  292. 		video::SColor(0, 255, 255, 255), video::SColor(0, 255, 255, 255),
  293. 		400, 1100
  294. 		);
  295. 	em->setMinStartSize(core::dimension2d<f32>(30.0f, 40.0f));
  296. 	em->setMaxStartSize(core::dimension2d<f32>(30.0f, 40.0f));
  297. 	ps->setEmitter(em);
  298. 	em->drop();
  299. 	//添加控制器
  300. 	scene::IParticleAffector* paf = ps->createFadeOutParticleAffector();
  301. 	ps->addAffector(paf);
  302. 	paf->drop();
  303. 	//调整材质设置
  304. 	ps->setMaterialFlag(video::EMF_LIGHTING, false);
  305. 	ps->setMaterialFlag(video::EMF_ZWRITE_ENABLE, false);
  306. 	ps->setMaterialTexture(0, driver->getTexture("../media/fireball.bmp"));
  307. 	ps->setMaterialType(video::EMT_TRANSPARENT_ADD_COLOR);
  308.  
  309. 	MyEventReceiver receiver(room, guiev, driver);
  310. 	device->setEventReceiver(&receiver);
  311.  
  312.  
  313. 	int lastFPS = -1;
  314. 	while (device->run())
  315. 	{
  316. 		driver->beginScene(true, true, video::SColor(255, 100, 101, 140));
  317.  
  318. 		smgr->drawAll();
  319. 		guiev->drawAll();
  320.  
  321. 		driver->endScene();
  322.  
  323. 		int fps = driver->getFPS();
  324. 		if (lastFPS != fps)
  325. 		{
  326. 			core::stringw str = L"11PerPixelLighting [";
  327. 			str += driver->getName();
  328. 			str += "]FPS.",
  329. 				str += fps;
  330. 			device->setWindowCaption(str.c_str());
  331. 			lastFPS = fps;
  332. 		}
  333. 	}
  334. 	device->drop();
  335.  
  336. 	return 0;
  337. }
