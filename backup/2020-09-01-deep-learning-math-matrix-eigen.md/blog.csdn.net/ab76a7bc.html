  69. {
  70. 	float temp = 0;

  72. 	{
  73. 		for (int j = 0; j < NUM; j++)

  87. 	float x[NUM], y, z;
  88. 	float u[NUM] = { 1, 1, 1 };
  89. 	int i, j, n = NUM;

  99. 			u[i] = x[i];
  100. 		}
  101. 	} while (fabs(z - y) >= 0.0000005);
  102. 	return sqrt(y);
