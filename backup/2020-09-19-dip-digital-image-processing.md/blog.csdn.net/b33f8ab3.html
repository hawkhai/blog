Morphological Operations: A set of operations that process images based on shapes.Morphological operations apply a structuring element to an input image and generate an output image.

The most basic morphological operations are two: Erosion and Dilation. They have a wide array of uses, i.e.:

(1)、Removing noise.

(2)、Isolation of individual elements and joining disparate elements in an image.

(3)、Finding of intensity bumps or holes in an image.

数学形态学可以理解为一种滤波行为，因此也称为形态学滤波。滤波中用到的滤波器(kernal)，在形态学中称为结构元素。结构元素往往是由一个特殊的形状构成,如线条、矩形、圆等。

         开运算(open)：先腐蚀后膨胀的过程。开运算可以用来消除小黑点，在纤细点处分离物体、平滑较大物体的边界的同时并不明显改变其面积。

         闭运算(close)：先膨胀后腐蚀的过程。闭运算可以用来排除小黑洞。

         形态学梯度(morph-grad)：可以突出团块(blob)的边缘，保留物体的边缘轮廓。

         顶帽(top-hat)：将突出比原轮廓亮的部分。

         黑帽(black-hat)：将突出比原轮廓暗的部分。


目前fbc_cv库中支持uchar和float两种数据类型，经测试，与OpenCV3.1结果完全一致。

  1. // fbc_cv is free software and uses the same licence as OpenCV
  2. // Email: fengbingchun@163.com

  4. #ifndef FBC_CV_MORPHOLOGYEX_HPP_
  5. #define FBC_CV_MORPHOLOGYEX_HPP_
  6.  
  7. /* reference: include/opencv2/imgproc.hpp
  8.               modules/imgproc/src/morph.cpp
  9. */
  10.  
  11. #include <typeinfo>
  12. #include "erode.hpp"
  13. #include "dilate.hpp"
  14.  
  15. namespace fbc {
  16.  
  17. // perform advanced morphological transformations using an erosion and dilation as basic operations
  18. // In case of multi - channel images, each channel is processed independently.
  19. // morphologyEx can be applied several ( iterations ) times.
  20. // op ==> enum MorphTypes
  21. // support type: uchar/float, multi-channels
  22. template<typename _Tp, int chs>
  23. int morphologyEx(const Mat_<_Tp, chs>& src, Mat_<_Tp, chs>& dst, int op, const Mat_<uchar, 1>& kernel,
  24. 	Point anchor = Point(-1, -1), int iterations = 1, int borderType = BORDER_CONSTANT, const Scalar& borderValue = Scalar::all(DBL_MAX))
  25. {
  26. 	FBC_Assert(typeid(uchar).name() == typeid(_Tp).name() || typeid(float).name() == typeid(_Tp).name()); // uchar || float
  27. 	if (dst.empty()) {
  28. 		dst = Mat_<_Tp, chs>(src.rows, src.cols);
  29. 	} else {
  30. 		FBC_Assert(src.rows == dst.rows && src.cols == dst.cols);
  31. 	}
  32.  
  33. 	Mat_<uchar, 1> kernel_ = kernel;
  34. 	if (kernel_.empty()) {
  35. 		kernel_ = Mat_<uchar, 1>(3, 3);
  36. 		getStructuringElement(kernel_, MORPH_RECT, Size(3, 3), Point(1, 1));
  37. 	}
  38.  
  39. 	switch (op) {
  40. 		case MORPH_ERODE: {
  41. 			erode(src, dst, kernel_, anchor, iterations, borderType, borderValue);
  42. 			break;
  43. 		}
  44. 		case MORPH_DILATE: {
  45. 			dilate(src, dst, kernel_, anchor, iterations, borderType, borderValue);
  46. 			break;
  47. 		}
  48. 		case MORPH_OPEN: {
  49. 			erode(src, dst, kernel_, anchor, iterations, borderType, borderValue);
  50. 			dilate(dst, dst, kernel_, anchor, iterations, borderType, borderValue);
  51. 			break;
  52. 		}
  53. 		case CV_MOP_CLOSE: {
  54. 			dilate(src, dst, kernel_, anchor, iterations, borderType, borderValue);
  55. 			erode(dst, dst, kernel_, anchor, iterations, borderType, borderValue);
  56. 			break;
  57. 		}
  58. 		case CV_MOP_GRADIENT: {
  59. 			Mat_<_Tp, chs> temp(src.rows, src.cols);
  60. 			erode(src, temp, kernel_, anchor, iterations, borderType, borderValue);
  61. 			dilate(src, dst, kernel_, anchor, iterations, borderType, borderValue);
  62. 			dst -= temp;
  63. 			break;
  64. 		}
  65. 		case CV_MOP_TOPHAT: {
  66. 			Mat_<_Tp, chs> temp(src.rows, src.cols);
  67. 			if (src.data != dst.data)
  68. 				temp = dst;
  69. 			erode(src, temp, kernel_, anchor, iterations, borderType, borderValue);
  70. 			dilate(temp, temp, kernel_, anchor, iterations, borderType, borderValue);
  71. 			dst = src - temp;
  72. 			break;
  73. 		}
  74. 		case CV_MOP_BLACKHAT: {
  75. 			Mat_<_Tp, chs> temp(src.rows, src.cols);
  76. 			if (src.data != dst.data)
  77. 				temp = dst;
  78. 			dilate(src, temp, kernel_, anchor, iterations, borderType, borderValue);
  79. 			erode(temp, temp, kernel_, anchor, iterations, borderType, borderValue);
  80. 			dst = temp - src;
  81. 			break;
  82. 		}
  83. 		case MORPH_HITMISS: {
  84. 			FBC_Assert(typeid(uchar).name() == typeid(_Tp).name() && chs == 1);
  85. 			Mat_<uchar, 1> k1 = (kernel_ == Mat_<uchar, 1>(kernel_.rows, kernel_.cols, Scalar::all(1)));
  86. 			Mat_<uchar, 1> k2 = (kernel_ == Mat_<int, 1>(kernel_.rows, kernel_.cols, Scalar::all(-1)));
  87. 			Mat_<_Tp, chs> e1, e2;
  88.  
  89. 			if (countNonZero(k1) <= 0)
  90. 				e1 = src;
  91. 			else
  92. 				erode(src, e1, k1, anchor, iterations, borderType, borderValue);
  93. 			if (countNonZero(k2) <= 0) {
  94. 				e2 = src;
  95. 			} else {
  96. 				Mat_<_Tp, chs> src_complement;
  97. 				bitwise_not(src, src_complement);
  98. 				erode(src_complement, e2, k2, anchor, iterations, borderType, borderValue);
  99. 			}
  100. 			bitwise_and(e1, e2, dst);
  101. 			break;
  102. 		}
  103. 		default:
  104. 			FBC_Assert("unknown morphological operation");
  105. 	}
  106.  
  107. 	return 0;
  108. }
  109.  
  110. } // namespace fbc
  111.  
  112. #endif // FBC_CV_MORPHOLOGYEX_HPP_
